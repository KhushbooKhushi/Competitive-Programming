//User function template for C++
class Solution{
public:	
	// Function returns the second
	// largest elements
	int print2largest(int arr[], int n) {
	    // code here
	    int f=-1,s=-1;
	    if(n>1){
	        if(arr[0]>arr[1]){
	            f = arr[0];
	            s=arr[1];
	        }
	        else if(arr[1]>arr[0])
	        {
	            f=arr[1];
	            s=arr[0];
	        }
	        
	       
	        for(int i=2;i<n;i++){
	           if(arr[i]>f){
	               s = f;
	               f = arr[i];
	           } 
	           else if(arr[i]>s && arr[i]<f){
	               s = arr[i];
	           }
	        }
	       
	       if(f == s)
	       {
	           s =-1;
	       }
	    }
	    return s;
	}
};

        int n;

        int arr[n];
        for (int i = 0; i < n; i++) {
            cin >> arr[i];
        }
        Solution ob;
        auto ans = ob.print2largest(arr, n);
        cout << ans << "\n";
    }
    return 0;
}
