class Solution
{
  public:
    vector<int> findIndex(int a[], int n, int key)
    {
        //code here.
        vector<int> v1 = {-1,-1};
        for(int i=0;i<n;i++){
            //first index from left
            if(a[i] == key && v1[0] == -1){
                v1[0] = i;
            }
            
            //first index from right
            if(a[i] == key){
                v1[1]=i;
            }
        }
        
        //for(int i=(n-1);i>=v1[0];i--){
          //  if(a[i] == key)
            //{
              //  v1[1] = i;
                //break;
            //}
        //}
        
        return v1;
    }
  
};
