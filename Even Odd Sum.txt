Even Odd Sum
Given an array Arr[] of N integers.Find the sum of values of even and odd index positions separately.


//{ Driver Code Starts
#include <bits/stdc++.h>
using namespace std;

// } Driver Code Ends
class Solution {
  public:
    vector<int> EvenOddSum(int N, int Arr[]) {
        // code here
        vector<int> v1={0,0};
        
        for(int i=0;i<N;i++){
            if(i%2==0){
                //odd numbers
                v1[0]+=Arr[i];
            }
            else{
                v1[1]+= Arr[i];
            }
        }
        return v1;
    }
};

//{ Driver Code Starts.
int main() {
    int t;
    cin >> t;
    while (t--) {
        int N;
        cin >> N;
        int Arr[N];
        for (int i = 0; i < N; i++) cin >> Arr[i];
        Solution ob;
        vector<int> ans = ob.EvenOddSum(N, Arr);
        cout << ans[0] << " " << ans[1] << "\n";
    }
}
